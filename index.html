# middlewares/subscription.py class SubscriptionMiddleware(BaseMiddleware): async def __call__(self, handler, event, data): bot = data["bot"] user_id = event.from_user.id member = await bot.get_chat_member( chat_id=settings.CHANNEL_ID, user_id=user_id, ) data["is_subscribed"] = member.status in ( "member", "administrator", "creator" ) return await handler(event, data)
